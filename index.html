module self
  # A definition of factorial, using a local, tail recursive function
  def factorial(n: Int): Int
    def go(n: Int, acc: Int): Int
      if (n <= 0)
        acc
      else
        go(n-1, n*acc)
      end
    end

    go(n, 1)
  end

  # Another implementation of `factorial`, this time with a `while` loop
  def factorial2(n: Int): Int
    def acc = 1
    def i = n

    while (i > 0)
      acc *= i
      i -= 1
    end

    acc
  end

  def compute(n: Int, f: Int -> Int): Int => f(n)

  def main()
    def number = 10 # Compute factorials for number 10

    trace(compute(number, factorial)) # Compute with recursive factorial method
    trace(compute(number, factorial2)) # Compute with loop factorial method
    # Result should be 3628800 in both
  end
end

package;using Lambda;using StringTools;/** 
We will use module here, because in modules, everything is static
and that is what we need for functional programming
 **/
class Main{
  // A definition of factorial, using a local, tail recursive function
  public static function factorial(n: Int): Int return{
    function go(n: Int, acc: Int): Int return{
      if (n <= 0){
        acc;
      }else{
        go(n-1, n*acc);
      }
    }

    go(n, 1);
  }

  // Another implementation of `factorial`, this time with a `while` loop
  public static function factorial2(n: Int): Int return{
    var acc = 1;
    var i = n;

    while (i > 0){
      acc *= i;
      i -= 1;
    }

    acc;
  }

  public static function compute(n: Int, f: Int -> Int): Int return f(n);

  public static function main() return{
    var number = 10 ;// Compute factorials for number 10

    trace(compute(number, factorial)) ;// Compute with recursive factorial method
    trace(compute(number, factorial2)) ;// Compute with loop factorial method
    // Result should be 3628800 in both
  }
}
